132 // Fast conversion from a IEEE 32-bit floating point number F in [0,1] to a
133 // a 32-bit integer I in [0,2^L-1].
134 //
135 // fFloat = F
136 // iLog = L
137 // iInt = I
138 
139 #define WM3_SCALED_FLOAT_TO_INT(fFloat,iLog,iInt)\
140 { \
141  int iShift = 150 - iLog - ((*(int*)(&fFloat) >> 23) & 0xFF); \
142  if (iShift < 24) \
143  { \
144  iInt = ((*(int*)(&fFloat) & 0x007FFFFF) | \
145  0x00800000) >> iShift; \
146  if (iInt == (1 << iLog)) \
147  { \
148  iInt--; \
149  } \
150  } \
151  else \
152  { \
153  iInt = 0; \
154  } \
155 }
