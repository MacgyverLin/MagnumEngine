57  bool writeClass(OutputStream &os, const ClassDesc &classDesc) const
58  {
59  if(classDesc.hasSuper())
60  {
61  if(!writeClass(os, classDesc.getSuper()))
62  return false;
63  }
64 
65  String className(classDesc.getName());
66  os << className;
67 
68  const ClassDesc::PropertyDescList &propertyList = classDesc.getProperties();
69 
70  int numProperties = propertyList.size();
71  os << numProperties;
72 
73  for(ClassDesc::PropertyDescListCItr itr=propertyList.begin(); itr!=propertyList.end(); itr++)
74  {
75  int classNameIndex = itr->getClassIndex();
76  String propertyName = itr->getName();
77 
78  assert(classNameIndex>=0 && classNameIndex<ClassIndexMap::instance().getNumClasses());
79  os << classNameIndex;
80  os << propertyName;
81 
82  if(!writePropertyDescFuncs[classNameIndex](os, *itr, reflectClass))
83  return false;
84  }
85 
86  return true;
87  }
88 private:
89  static bool (*writePropertyDescFuncs[])(OutputStream &os, const PropertyDesc &propertyDesc, const ReflectClass *ptr);
90 
91  template <class T>
92  static bool writePropertyDesc(OutputStream &os, const PropertyDesc &propertyDesc, const ReflectClass *ptr)
93  {
94  T val;
95 
96  propertyDesc.get(val, ptr);
97 
98  os << val;
99 
100  return true;
101  }
102 
103  const ReflectClass *reflectClass;
104 };
